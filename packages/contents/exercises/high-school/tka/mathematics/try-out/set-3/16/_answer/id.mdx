Maka penyelesaiannya <InlineMath math="\{x < 0 \cup x > 1\}" />. Sedangkan solusi umum dari pertidaksamannya adalah

<MathContainer>
<BlockMath math="\log_2(x^2 - x) \leq 1" />

<BlockMath math="\log_2(x^2 - x) \leq \log_2 2" />

<BlockMath math="x^2 - x \leq 2" />

<BlockMath math="x^2 - x - 2 \leq 0" />

<BlockMath math="(x + 1)(x - 2) \leq 0" />

<BlockMath math="x = -1 \cup x = 2" />
</MathContainer>
