Kita substitusikan nilai <InlineMath math="w = 5" />, <InlineMath math="z = 4" />, <InlineMath math="x = 3" />, dan <InlineMath math="y = 2" />.

<MathContainer>
  <BlockMath math="A(5 \otimes 4 \oslash 3 \oplus 2) = -5 + \frac{4}{2} + 3 \cdot 3" />
