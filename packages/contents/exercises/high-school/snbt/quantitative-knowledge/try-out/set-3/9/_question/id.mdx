<LineEquation
  title="Ilustrasi Lingkaran"
  description={<>Lingkaran dengan pusat <InlineMath math="O" /> dan sudut <InlineMath math="\angle BOC = 50^\circ" />.</>}
  cameraPosition={[0, 0, 12]}
  showZAxis={false}
  data={[
    {
      // Lingkaran
      points: Array.from({ length: 65 }, (_, i) => {
        const angle = (i / 64) * 2 * Math.PI;
        return {
          x: 3 * Math.cos(angle),
          y: 3 * Math.sin(angle),
          z: 0,
        };
      }),
      color: getColor("CYAN"),
      showPoints: false,
    },
    {
      // Garis AB (Diameter)
      points: Array.from({ length: 11 }, (_, i) => {
        const t = i / 10;
        return {
          x: -3 + t * 6,
          y: 0,
          z: 0,
        };
      }),
      color: getColor("YELLOW"),
      showPoints: false,
      labels: [
        { text: "A", at: 0, offset: [-0.5, 0.5, 0] },
        { text: "O", at: 5, offset: [0, 0.5, 0] },
        { text: "B", at: 10, offset: [0.5, 0.5, 0] },
      ],
    },
    {
      // Garis OC (Jari-jari)
      points: Array.from({ length: 11 }, (_, i) => {
        const t = i / 10;
        const angle = (-50 * Math.PI) / 180;
        return {
          x: t * 3 * Math.cos(angle),
          y: t * 3 * Math.sin(angle),
          z: 0,
        };
      }),
      color: getColor("ORANGE"),
      showPoints: false,
      labels: [
        { text: "C", at: 10, offset: [0.3, -0.4, 0] },
      ],
    },
    {
      // Garis AC (Tali busur)
      points: Array.from({ length: 11 }, (_, i) => {
        const t = i / 10;
        const ax = -3;
        const ay = 0;
        const angle = (-50 * Math.PI) / 180;
        const cx = 3 * Math.cos(angle);
        const cy = 3 * Math.sin(angle);
        
        return {
          x: ax + t * (cx - ax),
          y: ay + t * (cy - ay),
          z: 0,
        };
      }),
      color: getColor("PURPLE"),
      showPoints: false,
    },
    {
      // Arc untuk sudut 50 derajat (dari OB ke OC)
      points: Array.from({ length: 11 }, (_, i) => {
        const startAngle = 0; // Sudut untuk OB (sumbu x positif)
        const endAngle = (-50 * Math.PI) / 180; // Sudut untuk OC
        const angle = startAngle + (i / 10) * (endAngle - startAngle);
        const radius = 0.8; // Radius arc lebih kecil dari lingkaran
        return {
          x: radius * Math.cos(angle),
          y: radius * Math.sin(angle),
          z: 0,
        };
      }),
      color: getColor("VIOLET"),
      showPoints: false,
