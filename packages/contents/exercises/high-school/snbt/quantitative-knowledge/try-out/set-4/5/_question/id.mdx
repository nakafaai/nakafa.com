<LineEquation
  title="Kurva Parabola"
  description={<>Visualisasi kurva parabola pada bidang kartesius.</>}
  data={[
    {
      points: Array.from({ length: 61 }, (_, i) => {
        const y = -2 + (i / 60) * 6; // y from -2 to 4
        const x = -Math.pow(y - 1, 2) + 4;
        return { x, y, z: 0 };
      }),
      color: getColor("PURPLE"),
      showPoints: false,
      labels: [

      ],
    },
    {
      points: [

      ],
      color: getColor("TEAL"), // Helper line
      showPoints: false,
      labels: [
        { text: "1", at: 0, offset: [-0.5, 0, 0], color: getColor("INDIGO") },
      ],
    },
    {
      points: [

        { x: 4, y: 1, z: 0 },

      showPoints: false,
      labels: [
        { text: "4", at: 0, offset: [0, -0.5, 0], color: getColor("INDIGO") },
      ],
    },
  ]}

/>

2.  <InlineMath math="f(y) = -(y - 1)^2 + 4" />

4.  absis titik puncak adalah <InlineMath math="x = 4" />
