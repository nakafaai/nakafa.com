# Task 1.2: Create Audio Studies Queries

## Goal
Create queries for audio status and user history.

## File
`packages/backend/convex/audioStudies/queries.ts`

## Implementation

```typescript
import { query, internalQuery } from "@repo/backend/convex/functions";
import { vv } from "@repo/backend/convex/lib/validators";
import { audioStatusValidator } from "@repo/backend/convex/lib/contentValidators";
import { v } from "convex/values";

// Public: Get audio status (real-time)
export const getStatus = query({
  args: { id: vv.id("contentAudios") },
  returns: v.union(
    v.object({
      status: audioStatusValidator,
      audioUrl: v.optional(v.string()),
      duration: v.optional(v.number()),
      error: v.optional(v.string()),
    }),
    v.null()
  ),
  handler: async (ctx, args) => {
    const audio = await ctx.db.get("contentAudios", args.id);
    if (!audio) return null;
    const url = audio.status === "completed" && audio.audioStorageId
      ? await ctx.storage.getUrl(audio.audioStorageId)
      : undefined;
    return { 
      status: audio.status, 
      audioUrl: url || undefined, 
      duration: audio.audioDuration, 
      error: audio.errorMessage 
    };
  },
});

// Public: Get user listening history
export const getHistory = query({
  args: { limit: v.optional(v.number()) },
  returns: v.array(v.object({
    id: vv.id("userContentAudios"),
    contentId: v.union(vv.id("articleContents"), vv.id("subjectSections")),
    contentType: v.union(v.literal("article"), v.literal("subject")),
    title: v.string(),
    playCount: v.number(),
    lastPlayedAt: v.number(),
  })),
  handler: async (ctx, args) => {
    const identity = await ctx.auth.getUserIdentity();
    if (!identity) return [];
    
    const user = await ctx.db.query("users")
      .withIndex("tokenIdentifier", q => q.eq("tokenIdentifier", identity.tokenIdentifier))
      .first();
    if (!user) return [];

    const records = await ctx.db.query("userContentAudios")
      .withIndex("user", q => q.eq("userId", user._id))
      .order("desc")
      .take(args.limit || 20);
    
    return Promise.all(records.map(async r => {
      const content = r.contentType === "article"
        ? await ctx.db.get("articleContents", r.contentId)
        : await ctx.db.get("subjectSections", r.contentId);
      return {
        id: r._id,
        contentId: r.contentId,
        contentType: r.contentType,
        title: content?.title || "Unknown",
        playCount: r.playCount,
        lastPlayedAt: r.lastPlayedAt || r._creationTime,
      };
    }));
  },
});

// Internal: Get by ID
export const get = internalQuery({
  args: { id: vv.id("contentAudios") },
  returns: v.union(v.object({
    contentId: v.union(vv.id("articleContents"), vv.id("subjectSections")),
    contentType: v.union(v.literal("article"), v.literal("subject")),
    contentHash: v.string(),
    voiceId: v.string(),
    voiceSettings: v.optional(v.object({
      stability: v.optional(v.number()),
      similarityBoost: v.optional(v.number()),
      style: v.optional(v.number()),
      useSpeakerBoost: v.optional(v.boolean()),
    })),
    status: audioStatusValidator,
  }), v.null()),
  handler: async (ctx, args) => ctx.db.get("contentAudios", args.id),
});

// Internal: Get with script
export const getWithScript = internalQuery({
  args: { id: vv.id("contentAudios") },
  returns: v.union(v.object({
    script: v.string(),
    voiceId: v.string(),
    voiceSettings: v.optional(v.object({
      stability: v.optional(v.number()),
      similarityBoost: v.optional(v.number()),
      style: v.optional(v.number()),
      useSpeakerBoost: v.optional(v.boolean()),
    })),
  }), v.null()),
  handler: async (ctx, args) => {
    const a = await ctx.db.get("contentAudios", args.id);
    return a?.script ? { script: a.script, voiceId: a.voiceId, voiceSettings: a.voiceSettings } : null;
  },
});

// Internal: Find by content+voice
export const findByContentVoice = internalQuery({
  args: {
    contentId: v.union(vv.id("articleContents"), vv.id("subjectSections")),
    contentType: v.union(v.literal("article"), v.literal("subject")),
    voiceId: v.string(),
  },
  returns: v.union(v.object({
    id: vv.id("contentAudios"),
    contentHash: v.string(),
    status: audioStatusValidator,
    audioStorageId: v.optional(vv.id("_storage")),
  }), v.null()),
  handler: async (ctx, args) => {
    const a = await ctx.db.query("contentAudios")
      .withIndex("content_voice", q =>
        q.eq("contentId", args.contentId)
          .eq("contentType", args.contentType)
          .eq("voiceId", args.voiceId)
      ).first();
    return a ? { id: a._id, contentHash: a.contentHash, status: a.status, audioStorageId: a.audioStorageId } : null;
  },
});
```
