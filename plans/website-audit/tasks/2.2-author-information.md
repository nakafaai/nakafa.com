### Subtask 2.2.1: Create AuthorCard Component

**File to create**: `apps/www/components/contents/author-card.tsx`

```typescript
import { Avatar, AvatarFallback, AvatarImage } from "@repo/design-system/components/ui/avatar";
import Link from "next/link";

interface AuthorCardProps {
  name: string;
  bio?: string;
  image?: string;
  slug: string;
  locale: string;
  datePublished: string;
  dateModified?: string;
}

export function AuthorCard({
  name,
  bio,
  image,
  slug,
  locale,
  datePublished,
  dateModified,
}: AuthorCardProps) {
  const publishedDate = new Date(datePublished).toLocaleDateString(locale, {
    year: "numeric",
    month: "long",
    day: "numeric",
  });

  const modifiedDate = dateModified
    ? new Date(dateModified).toLocaleDateString(locale, {
        year: "numeric",
        month: "long",
        day: "numeric",
      })
    : null;

  return (
    <div className="flex items-start gap-4 border-t pt-6 mt-8">
      <Avatar className="h-12 w-12">
        <AvatarImage src={image} alt={name} />
        <AvatarFallback>{name.charAt(0)}</AvatarFallback>
      </Avatar>
      
      <div className="flex-1">
        <div className="flex items-center gap-2">
          <span className="text-sm text-muted-foreground">Written by</span>
          <Link
            href={`/${locale}/contributor/${slug}`}
            className="font-medium hover:underline"
          >
            {name}
          </Link>
        </div>
        
        {bio && (
          <p className="text-sm text-muted-foreground mt-1">{bio}</p>
        )}
        
        <div className="flex items-center gap-4 mt-2 text-xs text-muted-foreground">
          <time dateTime={datePublished}>Published: {publishedDate}</time>
          {modifiedDate && dateModified !== datePublished && (
            <time dateTime={dateModified}>Updated: {modifiedDate}</time>
          )}
        </div>
      </div>
    </div>
  );
}
```

2. **Bio**: Brief description (1-2 sentences)
